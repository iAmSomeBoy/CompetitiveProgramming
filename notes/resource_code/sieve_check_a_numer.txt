// Function to check if """""a number"""" is prime #not to do multiple number processing
bool isPrime(int num) {
    if (num <= 1) return false; //at first we assume all the number in the range are primes
    if (num == 2) return true;
    if (num % 2 == 0) return false;
    
    int limit = sqrt(num);
    for (int i = 3; i <= limit; i += 2) {
        if (num % i == 0) return false;
    }
    return true;
}
